3. Control Flow

If-Else

if 36 < 92
    println("36 is less than 92")
elseif 36 == 92
    println("36 equals 92")
else
    println("36 is greater than 92")
end

Loops

# For loop
for i in 1:10
    println("Number: $i")
end

# While loop
n = 3
while n > 0
    println("Countdown: $n")
    global n = n - 1
end

4. Multiple Dispatch

Julia calls different function implementations based on the types of the arguments.

function add(a::Int, b::Int)
    println("Adding Integers")
    a + b
end

function add(a::Float64, b::Float64)
    println("Adding Floats")
    a + b + 0.5
end

add(10, 20)      # Calls Int version
add(10.0, 20.0)  # Calls Float version
